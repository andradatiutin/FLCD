public static void SimonaHalep(){
    int number;
    int i;
    read(number);
    if(number%2==0 && number!=2){
        write("Number not prime!");
        stop;
    }
    if(number<=1){
        write("Number not prime!");
        stop;
    }
    i=3;
    while(i<=number/2){
        if(number%i==0){
            write("Number not prime");
            stop;
        }
    }
    write("Number prime");
}
